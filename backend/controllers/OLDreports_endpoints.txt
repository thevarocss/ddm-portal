// GetServiceRevenueReport returns JSON data for the revenue table
func GetServiceRevenueReport(pool *pgxpool.Pool) gin.HandlerFunc {
    return func(c *gin.Context) {
        from := c.Query("from")
        to := c.Query("to")

        rows, err := pool.Query(context.Background(), `
            SELECT day, total_charge_count, new_optins_revenue, renewal_revenue, total_revenue
            FROM mv_daily_revenue
            WHERE day BETWEEN $1 AND $2
            ORDER BY day DESC
        `, from, to)
        if err != nil {
            c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
            return
        }
        defer rows.Close()

        var results []map[string]interface{}
        for rows.Next() {
            var day time.Time
            var totalCharge, newRevenue, renewalRevenue, totalRevenue float64
            if err := rows.Scan(&day, &totalCharge, &newRevenue, &renewalRevenue, &totalRevenue); err != nil {
                c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
                return
            }
            results = append(results, gin.H{
                "Time":             day.Format("2006-01-02"),
                "TotalChargeCount": totalCharge,
                "NewOptInsRevenue": newRevenue,
                "RenewalRevenue":   renewalRevenue,
                "TotalRevenue":     totalRevenue,
            })
        }
        c.JSON(http.StatusOK, results)
    }
}

func GetOperationLogReport(pool *pgxpool.Pool) gin.HandlerFunc {
    return func(c *gin.Context) {
        from := c.Query("from")
        to := c.Query("to")
        msisdn := c.Query("msisdn")
        serviceId := c.Query("serviceId")
        productId := c.Query("productId")
        sequenceNo := c.Query("sequenceNo")

        rows, err := pool.Query(context.Background(), `
            SELECT time, msisdn, service_id, product_id, sequence_no, status, callback_type, message
            FROM ddm_callbacks
            WHERE time BETWEEN $1 AND $2
              AND ($3 = '' OR msisdn = $3)
              AND ($4 = '' OR service_id = $4)
              AND ($5 = '' OR product_id = $5)
              AND ($6 = '' OR sequence_no = $6)
            ORDER BY time DESC
        `, from, to, msisdn, serviceId, productId, sequenceNo)
        if err != nil {
            c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
            return
        }
        defer rows.Close()

        var results []map[string]interface{}
        for rows.Next() {
            var t time.Time
            var msisdnVal, serviceIdVal, productIdVal, sequenceNoVal, status, callbackType, message string
            if err := rows.Scan(&t, &msisdnVal, &serviceIdVal, &productIdVal, &sequenceNoVal, &status, &callbackType, &message); err != nil {
                c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
                return
            }
            results = append(results, gin.H{
                "Time":         t.Format("2006-01-02 15:04:05"),
                "MSISDN":       msisdnVal,
                "ServiceId":    serviceIdVal,
                "ProductId":    productIdVal,
                "SequenceNo":   sequenceNoVal,
                "Status":       status,
                "CallbackType": callbackType,
                "Message":      message,
            })
        }
        c.JSON(http.StatusOK, results)
    }
}
